Project Dojo API - version 1

# POST https://project-dojo.herokuapp.com/api/v1/register
Input:
{
  "name": "John Doe",
  "email": "jd@test.com"
  "password": "password",
  "role": "student"
}
Output:
{
  "value": "Success"
}
If error, it would return 404 or 500.

# POST https://project-dojo.herokuapp.com/api/v1/login
Input:
{
  "email": "jd@test.com"
  "password": "password",
  "role": "student"
}
Output:
{
  "value": "Success"
}
If error
{
  "value": "Fail"
}
